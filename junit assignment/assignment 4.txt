package junitbasics;

public class mMathUtils {

	public int add(int a,int b) {
		return a+b;
	}
	
	public double divide(int a, int b) {
		return a/b;
	}
	
	public int multiply(int a,int b) {
		return a*b;
	}
	
	public double computeCircle(double radius) {
		return Math.PI*radius*radius;
	}

}
